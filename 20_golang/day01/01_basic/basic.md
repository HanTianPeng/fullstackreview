### 位运算
60
60 / 2 = 30 --- 0       0           
30 / 2 = 15 --- 0       00          
15 / 2 = 7 ---- 1       100
7 / 2 = 3 ----- 1       1100
3 / 2 = 1 ----- 1       11100
1 / 2 = 0 ----- 1       111100  1*2^5 + 1*2^4 + 1*2^3 + 1*2^2 + 0*2^1 + 0*2^0 = 32 + 16 + 8 + 4 = 60

60 << 2 11110000 1*2^7 + 1*2^6 + 1*2^5 + 1*2^4 + 1*2^0 + 1*2^0 + 0*2^0 + 0*2^0 = 128 + 64 + 32 + 16 = 240

60 >> 2 1111 1*2^3 + 1*2^2 + 1*2^1 + 1*2^0 = 8 + 4 + 2 + 1 = 15
